#version  330 core

layout(location=0) in vec3 pos;
layout (location = 1) in vec2 aTexCoords;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec3 worldSpacePos;

void main() {
    gl_Position = projection * view * model * vec4(pos, 1);
    worldSpacePos = vec3(model * vec4(pos,1));
}
